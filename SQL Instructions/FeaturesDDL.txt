--TeamBigData.Utification.Features
CREATE TABLE dbo."Events"
(
userID int NOT NULL,
"disabled" int DEFAULT 0,
);

CREATE TABLE dbo.Pins
(
pinID int NOT NULL Identity,
userID int NOT NULL,
lat varchar(50),
lng varchar(50),
pinType int,
"description" varchar(MAX),
"disabled" int DEFAULT 0,
completed int DEFAULT 0,
"dateTime" varchar(50),
CONSTRAINT Pins_PK PRIMARY KEY (pinID)
);

CREATE TABLE dbo.Extensions
(
extension varchar(5) NOT NULL
CONSTRAINT Ext_PK PRIMARY KEY (extension)
);

INSERT INTO dbo.Extensions values ('.png'), ('.jpeg');

CREATE TABLE dbo.PinPic
(
pinID int,
"filename" varchar(50) NOT NULL,
extension varchar(5) NOT NULL,
CONSTRAINT PinPic_PK PRIMARY KEY(pinID),
CONSTRAINT PinPic_FK_01 Foreign KEY (pinID) references dbo.Pins(pinID),
CONSTRAINT PinPic_FK_02 Foreign Key (extension) references dbo.Extensions(extension),
CONSTRAINT PinPic_UK_01 UNIQUE ("filename")
);

CREATE TABLE dbo.ProfilePic
(
userID int,
"filename" varchar(50),
extension varchar(5),
PRIMARY KEY(userID),
CONSTRAINT ProfilePic_FK_01 Foreign Key (extension) references dbo.Extensions(extension),
CONSTRAINT ProfilePic_UK_01 UNIQUE ("filename")
);

CREATE TABLE dbo."Services"
(
userID int NOT NULL,
"disabled" int DEFAULT 0,
);
